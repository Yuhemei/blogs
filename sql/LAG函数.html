LAG 函数教程 ​

什么是 LAG 函数 ​

LAG 是窗口函数，用于访问当前行的前 N 行的值。不需要自连接或子查询。

基本语法 ​

sql
LAG(column, offset, default_value) 
OVER (
    PARTITION BY partition_column 
    ORDER BY order_column
)

参数说明：

  • column：要获取值的列
  • offset：向前偏移的行数（默认 1）
  • default_value：第一行返回的值（默认 NULL）
  • PARTITION BY：分组
  • ORDER BY：排序

基础示例 ​

sql
-- 示例：员工薪资表
CREATE TABLE employees (
    id INT,
    name VARCHAR(50),
    salary INT,
    hire_date DATE
);

INSERT INTO employees VALUES 
(1, 'Alice', 5000, '2020-01-01'),
(2, 'Bob', 6000, '2020-02-01'),
(3, 'Charlie', 5500, '2020-03-01'),
(4, 'David', 7000, '2020-04-01');

-- ✅ 基础用法：获取前一个月薪
SELECT 
    name,
    salary,
    LAG(salary, 1) OVER (ORDER BY hire_date) as prev_month_salary
FROM employees;

-- 结果：
-- name    | salary | prev_month_salary
-- Alice   | 5000   | NULL
-- Bob     | 6000   | 5000
-- Charlie | 5500   | 6000
-- David   | 7000   | 5500

分组使用（PARTITION BY） ​

sql
-- 按部门分组，获取同部门前一个员工的薪资
SELECT 
    name,
    department_id,
    salary,
    LAG(salary, 1) OVER (
        PARTITION BY department_id 
        ORDER BY hire_date
    ) as prev_department_salary
FROM employees;

-- 结果（假设有部门ID）：
-- name   | dept_id | salary | prev_department_salary
-- Alice  | 1       | 5000   | NULL
-- Bob    | 1       | 6000   | 5000
-- Charlie| 2       | 5500   | NULL
-- David  | 2       | 7000   | 5500

实际应用场景 ​

场景1：计算差值（气温问题） ​

sql
-- ✅ 解决之前的天气问题
SELECT 
    recordDate,
    temperature,
    LAG(temperature, 1) OVER (ORDER BY recordDate) as prev_temperature,
    temperature - LAG(temperature, 1) OVER (ORDER BY recordDate) as temp_diff
FROM Weather
ORDER BY recordDate;

-- 找出比前一天高的日期
SELECT 
    recordDate
FROM (
    SELECT 
        id,
        recordDate,
        temperature,
        LAG(temperature, 1) OVER (ORDER BY recordDate) as prev_temperature
    FROM Weather
) t
WHERE temperature > prev_temperature;

场景2：计算增长率 ​

sql
-- 计算环比增长率
SELECT 
    date,
    sales,
    LAG(sales, 1) OVER (ORDER BY date) as prev_sales,
    (sales - LAG(sales, 1) OVER (ORDER BY date)) * 100.0 / 
        LAG(sales, 1) OVER (ORDER BY date) as growth_rate
FROM Sales
ORDER BY date;

场景3：检测变化 ​

sql
-- 检测价格变化
SELECT 
    date,
    product_name,
    price,
    LAG(price, 1) OVER (PARTITION BY product_name ORDER BY date) as prev_price,
    CASE 
        WHEN price > LAG(price, 1) OVER (PARTITION BY product_name ORDER BY date) 
        THEN '涨价'
        WHEN price < LAG(price, 1) OVER (PARTITION BY product_name ORDER BY date) 
        THEN '降价'
        ELSE '持平'
    END as price_change
FROM Prices
ORDER BY product_name, date;

场景4：计算移动平均 ​

sql
-- 计算3期移动平均
SELECT 
    date,
    sales,
    LAG(sales, 1) OVER (ORDER BY date) as prev_sales,
    LAG(sales, 2) OVER (ORDER BY date) as prev_2_sales,
    (sales + LAG(sales, 1) OVER (ORDER BY date) + 
     LAG(sales, 2) OVER (ORDER BY date)) / 3.0 as moving_avg_3
FROM Sales
ORDER BY date;

LAG vs 自连接对比 ​

sql
-- ❌ 使用自连接（复杂且性能差）
SELECT 
    a.id,
    a.salary,
    b.salary as prev_salary
FROM employees a
LEFT JOIN employees b ON b.id = a.id - 1;  -- 依赖连续ID，不可靠

-- ✅ 使用LAG函数（简洁高效）
SELECT 
    id,
    salary,
    LAG(salary, 1) OVER (ORDER BY id) as prev_salary
FROM employees;

常见陷阱 ​

陷阱1：忘记 ORDER BY ​

sql
-- ❌ 错误：没有ORDER BY，结果不按预期
SELECT 
    name,
    salary,
    LAG(salary, 1) OVER () as prev_salary  -- 缺失ORDER BY！
FROM employees;

-- ✅ 正确：必须指定ORDER BY
SELECT 
    name,
    salary,
    LAG(salary, 1) OVER (ORDER BY hire_date) as prev_salary
FROM employees;

陷阱2：处理 NULL 值 ​

sql
-- ⚠️ 注意：第一行返回NULL
SELECT 
    date,
    value,
    LAG(value, 1) OVER (ORDER BY date) as prev_value
FROM table_name;
-- 第1行 prev_value = NULL

-- ✅ 解决方案1：设置默认值
SELECT 
    date,
    value,
    LAG(value, 1, 0) OVER (ORDER BY date) as prev_value
FROM table_name;

-- ✅ 解决方案2：过滤NULL
SELECT * FROM (
    SELECT 
        date,
        value,
        LAG(value, 1) OVER (ORDER BY date) as prev_value
    FROM table_name
) t
WHERE prev_value IS NOT NULL;

与 LEAD 函数对比 ​

sql
-- LAG: 向前看（获取前面的行）
SELECT 
    name,
    salary,
    LAG(salary, 1) OVER (ORDER BY id) as prev_salary,    -- 前一行
    salary                                              as current_salary,
    LEAD(salary, 1) OVER (ORDER BY id) as next_salary     -- 后一行
FROM employees;

-- 结果：
-- name | prev | current | next
-- Alice| NULL | 5000    | 6000
-- Bob  | 5000 | 6000    | 5500
-- Charlie|6000|5500 | 7000
-- David| 5500 | 7000    | NULL

练习题 ​

sql
-- 题目：找出连续两天价格上涨的产品
WITH price_changes AS (
    SELECT 
        product_id,
        date,
        price,
        LAG(price, 1) OVER (PARTITION BY product_id ORDER BY date) as prev_price,
        LAG(date, 1) OVER (PARTITION BY product_id ORDER BY date) as prev_date
    FROM products
)
SELECT 
    product_id,
    prev_date as yesterday,
    date as today,
    price - prev_price as increase
FROM price_changes
WHERE price > prev_price;

要点总结 ​

  1. LAG 访问前 N 行的值
  2. 必须使用 ORDER BY
  3. 可选 PARTITION BY 分组
  4. 第一行返回 NULL 或指定默认值
  5. 比自连接更简洁高效
  6. 可用于差值、增长率、移动平均等计算

LAG 适合计算差值、比较上下行、移动平均等需要“向前看”的场景。
