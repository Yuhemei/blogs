OVER 关键字的含义 ​

OVER 定义窗口函数的作用范围，即“对哪些数据计算”。

OVER 结构 ​

sql
窗口函数名(参数) OVER (
    [PARTITION BY 列名]    -- 可选：分组
    [ORDER BY 列名]        -- 可选：排序
    [ROWS BETWEEN ...]     -- 可选：定义窗口范围
)

OVER 的作用 ​

将聚合或排名应用到“窗口”内每一行，而不是整体。

sql
-- ❌ 没有OVER：普通聚合（所有行只返回1个结果）
SELECT 
    name,
    salary,
    SUM(salary) as total_salary  -- 返回1行
FROM employees;

-- 结果：
-- name | salary | total_salary
-- Alice| 5000   | 25000  (所有薪资总和)

-- ✅ 有OVER：窗口函数（每行都有结果）
SELECT 
    name,
    salary,
    SUM(salary) OVER () as running_total  -- 每行都返回相同值
FROM employees;

-- 结果：
-- name | salary | running_total
-- Alice| 5000   | 25000
-- Bob  | 6000   | 25000
-- Charlie|5500 | 25000
-- David| 7000   | 25000

三种常用形式 ​

1️⃣ OVER() - 整个表作为窗口 ​

sql
-- 对整张表计算
SELECT 
    name,
    salary,
    SUM(salary) OVER () as total_salary,
    AVG(salary) OVER () as avg_salary,
    COUNT(*) OVER () as total_count
FROM employees;

-- 结果：每行都有整体统计信息
-- name   | salary | total_salary | avg_salary | total_count
-- Alice  | 5000   | 25000        | 6250       | 4
-- Bob    | 6000   | 25000        | 6250       | 4
-- Charlie| 5500   | 25000        | 6250       | 4
-- David  | 7000   | 25000        | 6250       | 4

2️⃣ OVER(ORDER BY 列) - 按顺序累计 ​

sql
-- 累计求和（从第一行到当前行）
SELECT 
    name,
    salary,
    SUM(salary) OVER (ORDER BY salary) as cumulative_sum
FROM employees;

-- 结果：
-- name   | salary | cumulative_sum
-- Alice  | 5000   | 5000    (5000)
-- Charlie| 5500   | 10500   (5000+5500)
-- Bob    | 6000   | 16500   (5000+5500+6000)
-- David  | 7000   | 23500   (5000+5500+6000+7000)

3️⃣ OVER(PARTITION BY 列) - 分组计算 ​

sql
-- 按部门分组计算
SELECT 
    name,
    department_id,

-- 结果：

组合使用：PARTITION BY + ORDER BY ​

sql
-- 按部门分组，并按薪资累计
SELECT 
    name,
    department_id,
    salary,
    SUM(salary) OVER (
        PARTITION BY department_id 
        ORDER BY salary
    ) as dept_cumulative_sum
FROM employees;

-- 结果（部门1：5000+6000；部门2：5500+7000）
-- name   | dept_id | salary | dept_cumulative
-- Alice  | 1       | 5000   | 5000
-- Bob    | 1       | 6000   | 11000
-- Charlie| 2       | 5500   | 5500
-- David  | 2       | 7000   | 12500

对比普通聚合与窗口函数 ​

sql
-- 普通聚合函数（GROUP BY）
SELECT 
    department_id,
    SUM(salary) as total_salary
FROM employees
GROUP BY department_id;

-- 结果：每个部门只有1行
-- dept_id | total_salary
-- 1       | 11000
-- 2       | 12500

-- 窗口函数（OVER）
SELECT 
    name,
    department_id,
    salary,
    SUM(salary) OVER (PARTITION BY department_id) as dept_total
FROM employees;

-- 结果：每行都有详细信息，同时显示部门总计
-- name   | dept_id | salary | dept_total
-- Alice  | 1       | 5000   | 11000
-- Bob    | 1       | 6000   | 11000
-- Charlie| 2       | 5500   | 12500
-- David  | 2       | 7000   | 12500

在天气问题中的应用 ​

sql
-- 使用OVER来比较相邻日期
SELECT 
    recordDate,
    temperature,
    LAG(temperature, 1) OVER (ORDER BY recordDate) as prev_temperature,
    temperature - LAG(temperature, 1) OVER (ORDER BY recordDate) as diff
FROM Weather;

-- OVER(ORDER BY recordDate) 的作用：
-- 定义窗口：按日期排序的所有行
-- 对每一行，找它"前面一行"的温度

核心区别 ​

普通聚合函数 vs 窗口函数 ​

sql
-- ❌ 这样不行：混用普通聚合
SELECT 
    name,
    salary,
    SUM(salary)  -- 错误！需要GROUP BY或OVER
FROM employees;

-- ✅ 选项1：使用GROUP BY（每个部门1行）
SELECT 
    department_id,
    SUM(salary) 
FROM employees
GROUP BY department_id;

-- ✅ 选项2：使用OVER（每行都有结果）
SELECT 
    name,
    department_id,
    salary,
    SUM(salary) OVER (PARTITION BY department_id)
FROM employees;

OVER 子句的组成部分 ​

sql
SUM(salary) OVER (
    PARTITION BY department_id    -- ① 分组：按部门
    ORDER BY salary               -- ② 排序：按薪资
    ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW  -- ③ 范围：从开头到当前
)

各部分的含义 ​

部分作用示例
PARTITION BY分组每个部门独立计算
ORDER BY排序按薪资从小到大
ROWS BETWEEN范围从第一行到当前行

语法图 ​

sql
函数名(参数) OVER (
    [PARTITION BY 列1, 列2...]
    [ORDER BY 列1 [ASC|DESC], 列2...]
    [ROWS BETWEEN 起始 AND 结束]
)

-- 最简单的形式
SUM(salary) OVER ()

-- 最复杂的形式
AVG(salary) OVER (
    PARTITION BY department_id 
    ORDER BY hire_date 
    ROWS BETWEEN 2 PRECEDING AND CURRENT ROW
)

一句话总结 ​

  • 定义窗口范围
  • 决定按组、排序列还是跨行范围
  • 可保留明细行并同时计算聚合或排名
