1. 多表连接查询 ​

员工部门薪资统计 ​

sql
-- 查询每个部门的平均薪资，并显示部门名称和员工数量
SELECT 
    d.department_name,
    COUNT(e.employee_id) as employee_count,
    AVG(e.salary) as avg_salary,
    MAX(e.salary) as max_salary,
    MIN(e.salary) as min_salary
FROM departments d
LEFT JOIN employees e ON d.department_id = e.department_id
WHERE e.hire_date >= '2020-01-01'
GROUP BY d.department_id, d.department_name
HAVING COUNT(e.employee_id) > 5
ORDER BY avg_salary DESC;

2. 窗口函数应用 ​

员工薪资排名 ​

sql
-- 查询每个部门内员工的薪资排名
SELECT 
    employee_id,
    first_name,
    last_name,
    department_id,
    salary,
    ROW_NUMBER() OVER (PARTITION BY department_id ORDER BY salary DESC) as salary_rank,
    RANK() OVER (PARTITION BY department_id ORDER BY salary DESC) as salary_rank_with_ties,
    DENSE_RANK() OVER (PARTITION BY department_id ORDER BY salary DESC) as dense_rank,
    LAG(salary, 1) OVER (PARTITION BY department_id ORDER BY salary DESC) as prev_salary,
    LEAD(salary, 1) OVER (PARTITION BY department_id ORDER BY salary DESC) as next_salary,
    salary - LAG(salary, 1) OVER (PARTITION BY department_id ORDER BY salary DESC) as salary_diff
FROM employees
WHERE department_id IS NOT NULL
ORDER BY department_id, salary DESC;

3. 递归CTE查询 ​

组织架构层级查询 ​

sql
-- 查询员工的管理层级关系
WITH RECURSIVE employee_hierarchy AS (
    -- 基础查询：顶级管理者
    SELECT 
        employee_id,
        first_name,
        last_name,
        manager_id,
        1 as level,
        CAST(first_name || ' ' || last_name AS VARCHAR(1000)) as hierarchy_path
    FROM employees 
    WHERE manager_id IS NULL
    
    UNION ALL
    
    -- 递归查询：下级员工
    SELECT 
        e.employee_id,
        e.first_name,
        e.last_name,
        e.manager_id,
        eh.level + 1,
        eh.hierarchy_path || ' -> ' || e.first_name || ' ' || e.last_name
    FROM employees e
    INNER JOIN employee_hierarchy eh ON e.manager_id = eh.employee_id
    WHERE eh.level < 10  -- 防止无限递归
)
SELECT 
    employee_id,
    first_name,
    last_name,
    level,
    hierarchy_path
FROM employee_hierarchy
ORDER BY level, first_name;

4. 复杂子查询 ​

销售业绩分析 ​

sql
-- 查询销售额超过部门平均值的销售员
SELECT 
    s.salesperson_id,
    s.first_name,
    s.last_name,
    s.department_id,
    SUM(o.order_amount) as total_sales,
    COUNT(o.order_id) as order_count,
    AVG(o.order_amount) as avg_order_amount
FROM salespeople s
INNER JOIN orders o ON s.salesperson_id = o.salesperson_id
WHERE o.order_date >= '2023-01-01'
    AND s.salesperson_id IN (
        -- 子查询：找出销售额超过部门平均值的销售员
        SELECT s2.salesperson_id
        FROM salespeople s2
        INNER JOIN orders o2 ON s2.salesperson_id = o2.salesperson_id
        WHERE o2.order_date >= '2023-01-01'
        GROUP BY s2.salesperson_id, s2.department_id
        HAVING SUM(o2.order_amount) > (
            -- 嵌套子查询：计算部门平均销售额
            SELECT AVG(dept_sales.total_sales)
            FROM (
                SELECT SUM(o3.order_amount) as total_sales
                FROM salespeople s3
                INNER JOIN orders o3 ON s3.salesperson_id = o3.salesperson_id
                WHERE s3.department_id = s2.department_id
                    AND o3.order_date >= '2023-01-01'
                GROUP BY s3.salesperson_id
            ) dept_sales
        )
    )
GROUP BY s.salesperson_id, s.first_name, s.last_name, s.department_id
ORDER BY total_sales DESC;

5. 动态分组和透视 ​

月度销售报表 ​

sql
-- 生成月度销售透视表
SELECT 
    product_category,
    SUM(CASE WHEN EXTRACT(MONTH FROM order_date) = 1 THEN order_amount ELSE 0 END) as jan_sales,
    SUM(CASE WHEN EXTRACT(MONTH FROM order_date) = 2 THEN order_amount ELSE 0 END) as feb_sales,
    SUM(CASE WHEN EXTRACT(MONTH FROM order_date) = 3 THEN order_amount ELSE 0 END) as mar_sales,
    SUM(CASE WHEN EXTRACT(MONTH FROM order_date) = 4 THEN order_amount ELSE 0 END) as apr_sales,
    SUM(CASE WHEN EXTRACT(MONTH FROM order_date) = 5 THEN order_amount ELSE 0 END) as may_sales,
    SUM(CASE WHEN EXTRACT(MONTH FROM order_date) = 6 THEN order_amount ELSE 0 END) as jun_sales,
    SUM(order_amount) as total_sales,
    COUNT(order_id) as total_orders
FROM orders o
INNER JOIN products p ON o.product_id = p.product_id
WHERE EXTRACT(YEAR FROM order_date) = 2023
GROUP BY product_category
ORDER BY total_sales DESC;

6. 复杂条件查询 ​

客户行为分析 ​

sql
-- 查询高价值客户及其购买行为
SELECT 
    c.customer_id,
    c.customer_name,
    c.email,
    COUNT(DISTINCT o.order_id) as total_orders,
    SUM(o.order_amount) as total_spent,
    AVG(o.order_amount) as avg_order_value,
    MAX(o.order_date) as last_order_date,
    MIN(o.order_date) as first_order_date,
    DATEDIFF(MAX(o.order_date), MIN(o.order_date)) as customer_lifespan_days,
    -- 计算客户价值评分
    CASE 
        WHEN SUM(o.order_amount) > 10000 THEN 'VIP'
        WHEN SUM(o.order_amount) > 5000 THEN 'Gold'
        WHEN SUM(o.order_amount) > 1000 THEN 'Silver'
        ELSE 'Bronze'
    END as customer_tier,
    -- 计算购买频率
    CASE 
        WHEN COUNT(DISTINCT o.order_id) > 20 THEN 'High Frequency'
        WHEN COUNT(DISTINCT o.order_id) > 10 THEN 'Medium Frequency'
        ELSE 'Low Frequency'
    END as purchase_frequency
FROM customers c
INNER JOIN orders o ON c.customer_id = o.customer_id
WHERE o.order_date >= DATE_SUB(CURRENT_DATE, INTERVAL 12 MONTH)
    AND o.order_status = 'completed'
    AND c.customer_id IN (
        -- 子查询：排除有退货记录的客户
        SELECT DISTINCT customer_id 
        FROM orders 
        WHERE order_status = 'returned'
    )
GROUP BY c.customer_id, c.customer_name, c.email
HAVING COUNT(DISTINCT o.order_id) >= 3
    AND SUM(o.order_amount) >= 1000
ORDER BY total_spent DESC, total_orders DESC;

7. 数据质量检查 ​

数据完整性验证 ​

sql
-- 检查数据完整性和一致性
WITH data_quality_check AS (
    SELECT 
        'Missing Email' as check_type,
        COUNT(*) as issue_count,
        'customers' as table_name
    FROM customers 
    WHERE email IS NULL OR email = ''
    
    UNION ALL
    
    SELECT 
        'Invalid Phone' as check_type,
        COUNT(*) as issue_count,
        'customers' as table_name
    FROM customers 
    WHERE phone NOT REGEXP '^[0-9]{10,15}$'
    
    UNION ALL
    
    SELECT 
        'Future Birth Date' as check_type,
        COUNT(*) as issue_count,
        'customers' as table_name
    FROM customers 
    WHERE birth_date > CURRENT_DATE
    
    UNION ALL
    
    SELECT 
        'Orphaned Orders' as check_type,
        COUNT(*) as issue_count,
        'orders' as table_name
    FROM orders o
    LEFT JOIN customers c ON o.customer_id = c.customer_id
    WHERE c.customer_id IS NULL
    
    UNION ALL
    
    SELECT 
        'Negative Amount' as check_type,
        COUNT(*) as issue_count,
        'orders' as table_name
    FROM orders 
    WHERE order_amount < 0
)
SELECT 
    check_type,
    issue_count,
    table_name,
    CASE 
        WHEN issue_count = 0 THEN 'PASS'
        WHEN issue_count < 10 THEN 'WARNING'
        ELSE 'FAIL'
    END as status
FROM data_quality_check
ORDER BY issue_count DESC;

8. 性能优化查询 ​

分页查询优化 ​

sql
-- 优化的分页查询，避免OFFSET性能问题
WITH ranked_orders AS (
    SELECT 
        o.order_id,
        o.customer_id,
        o.order_date,
        o.order_amount,
        o.order_status,
        c.customer_name,
        ROW_NUMBER() OVER (ORDER BY o.order_date DESC, o.order_id DESC) as row_num
    FROM orders o
    INNER JOIN customers c ON o.customer_id = c.customer_id
    WHERE o.order_date >= '2023-01-01'
        AND o.order_status IN ('completed', 'shipped')
)
SELECT 
    order_id,
    customer_id,
    customer_name,
    order_date,
    order_amount,
    order_status
FROM ranked_orders
WHERE row_num BETWEEN 1001 AND 1100  -- 第11页，每页100条
ORDER BY order_date DESC, order_id DESC;

9. 复杂聚合查询 ​

移动平均计算 ​

sql
-- 计算产品销量的移动平均
SELECT 
    product_id,
    product_name,
    sale_date,
    daily_sales,
    -- 7天移动平均
    AVG(daily_sales) OVER (
        PARTITION BY product_id 
        ORDER BY sale_date 
        ROWS BETWEEN 6 PRECEDING AND CURRENT ROW
    ) as moving_avg_7d,
    -- 30天移动平均
    AVG(daily_sales) OVER (
        PARTITION BY product_id 
        ORDER BY sale_date 
        ROWS BETWEEN 29 PRECEDING AND CURRENT ROW
    ) as moving_avg_30d,
    -- 累计销量
    SUM(daily_sales) OVER (
        PARTITION BY product_id 
        ORDER BY sale_date 
        ROWS UNBOUNDED PRECEDING
    ) as cumulative_sales,
    -- 销量排名
    RANK() OVER (
        PARTITION BY sale_date 
        ORDER BY daily_sales DESC
    ) as daily_rank
FROM (
    SELECT 
        p.product_id,
        p.product_name,
        DATE(o.order_date) as sale_date,
        SUM(o.quantity) as daily_sales
    FROM orders o
    INNER JOIN order_items oi ON o.order_id = oi.order_id
    INNER JOIN products p ON oi.product_id = p.product_id
    WHERE o.order_date >= '2023-01-01'
        AND o.order_status = 'completed'
    GROUP BY p.product_id, p.product_name, DATE(o.order_date)
) daily_sales
ORDER BY product_id, sale_date;

10. 复杂业务逻辑查询 ​

客户流失预测 ​

sql
-- 识别可能流失的客户
WITH customer_metrics AS (
    SELECT 
        c.customer_id,
        c.customer_name,
        COUNT(DISTINCT o.order_id) as total_orders,
        SUM(o.order_amount) as total_spent,
        AVG(o.order_amount) as avg_order_value,
        MAX(o.order_date) as last_order_date,
        MIN(o.order_date) as first_order_date,
        DATEDIFF(CURRENT_DATE, MAX(o.order_date)) as days_since_last_order,
        DATEDIFF(MAX(o.order_date), MIN(o.order_date)) as customer_lifespan_days,
        -- 计算购买频率（订单数/天数）
        COUNT(DISTINCT o.order_id) / NULLIF(DATEDIFF(MAX(o.order_date), MIN(o.order_date)), 0) as purchase_frequency
    FROM customers c
    INNER JOIN orders o ON c.customer_id = o.customer_id
    WHERE o.order_status = 'completed'
    GROUP BY c.customer_id, c.customer_name
),
customer_segments AS (
    SELECT 
        *,
        CASE 
            WHEN days_since_last_order > 90 THEN 'At Risk'
            WHEN days_since_last_order > 60 THEN 'Warning'
            WHEN days_since_last_order > 30 THEN 'Attention'
            ELSE 'Active'
        END as risk_level,
        CASE 
            WHEN total_spent > 5000 AND purchase_frequency > 0.1 THEN 'VIP'
            WHEN total_spent > 1000 AND purchase_frequency > 0.05 THEN 'Loyal'
            WHEN total_spent > 500 THEN 'Regular'
            ELSE 'New'
        END as customer_segment
    FROM customer_metrics
)
SELECT 
    customer_id,
    customer_name,
    total_orders,
    total_spent,
    avg_order_value,
    last_order_date,
    days_since_last_order,
    purchase_frequency,
    risk_level,
    customer_segment,
    -- 流失概率评分
    CASE 
        WHEN days_since_last_order > 90 AND purchase_frequency < 0.02 THEN 'High Risk'
        WHEN days_since_last_order > 60 AND purchase_frequency < 0.05 THEN 'Medium Risk'
        WHEN days_since_last_order > 30 AND purchase_frequency < 0.1 THEN 'Low Risk'
        ELSE 'Low Risk'
    END as churn_probability
FROM customer_segments
WHERE risk_level IN ('At Risk', 'Warning')
    OR (customer_segment = 'VIP' AND days_since_last_order > 30)
ORDER BY 
    CASE churn_probability 
        WHEN 'High Risk' THEN 1 
        WHEN 'Medium Risk' THEN 2 
        ELSE 3 
    END,
    total_spent DESC;
