SQL 函数使用指南 ​

基于现有文档，SQL 函数主要有三类：

SQL 函数分类 ​

1️⃣ 聚合函数（Aggregate Functions） ​

对一组值进行计算，返回单个值

2️⃣ 标量函数（Scalar Functions） ​

对单个值进行计算，每行返回一个结果

3️⃣ 窗口函数（Window Functions） ​

在数据行的指定窗口内执行计算

一、聚合函数 ​

常用聚合函数 ​

sql
-- COUNT - 计数
SELECT COUNT(*) FROM users;                    -- 所有行数
SELECT COUNT(DISTINCT age) FROM users;         -- 去重计数
SELECT COUNT(phone) FROM users;                -- 非NULL计数

-- SUM - 求和
SELECT SUM(amount) FROM orders;
SELECT SUM(amount) FROM orders WHERE status = 'completed';

-- AVG - 平均值
SELECT AVG(price) FROM products;
SELECT ROUND(AVG(price), 2) FROM products;     -- 保留2位小数

-- MAX/MIN - 最值
SELECT MAX(price), MIN(price) FROM products;
SELECT MAX(salary), MIN(salary) FROM employees WHERE department = 'IT';

-- 组合使用
SELECT 
    COUNT(*) as total,
    SUM(amount) as total_amount,
    AVG(amount) as avg_amount,
    MAX(amount) as max_amount,
    MIN(amount) as min_amount
FROM orders;

与 GROUP BY 配合 ​

sql
-- 按部门统计
SELECT 
    department_id,
    COUNT(*) as employee_count,
    AVG(salary) as avg_salary,
    MAX(salary) as max_salary
FROM employees
GROUP BY department_id;

-- 多个分组
SELECT 
    department_id,
    job_title,
    COUNT(*) as count,
    AVG(salary) as avg_salary
FROM employees
GROUP BY department_id, job_title;

-- 与 HAVING 配合过滤分组
SELECT 
    department_id,
    COUNT(*) as count
FROM employees
GROUP BY department_id
HAVING COUNT(*) > 5;

二、标量函数 ​

字符串函数 ​

sql
-- 连接字符串
SELECT CONCAT(first_name, ' ', last_name) as full_name FROM users;
SELECT first_name || ' ' || last_name as full_name FROM users;  -- PostgreSQL

-- 字符串长度
SELECT LENGTH(name) FROM users;
SELECT LEN(name) FROM users;  -- SQL Server

-- 大小写转换
SELECT UPPER(name), LOWER(email) FROM users;

-- 截取字符串
SELECT SUBSTRING(name, 1, 3) FROM users;       -- 前3个字符
SELECT LEFT(name, 3) FROM users;                -- SQL Server/MySQL
SELECT RIGHT(name, 3) FROM users;

-- 替换
SELECT REPLACE(phone, '-', '') FROM users;

-- 去除空格
SELECT TRIM(name) FROM users;
SELECT LTRIM(name), RTRIM(name) FROM users;

日期函数 ​

sql
-- 当前日期时间
SELECT CURRENT_DATE, CURRENT_TIMESTAMP;
SELECT NOW();  -- MySQL
SELECT GETDATE();  -- SQL Server

-- 提取日期部分
SELECT 
    EXTRACT(YEAR FROM date) as year,
    EXTRACT(MONTH FROM date) as month,
    EXTRACT(DAY FROM date) as day
FROM orders;

-- 日期计算
SELECT date + INTERVAL '1 day';            -- PostgreSQL
SELECT DATE_ADD(date, INTERVAL 1 DAY);     -- MySQL
SELECT DATEADD(day, 1, date);              -- SQL Server

-- 日期差
SELECT date1 - date2;                      -- PostgreSQL（返回天数）
SELECT DATEDIFF(date1, date2);             -- MySQL/SQL Server

-- 格式化日期
SELECT TO_CHAR(date, 'YYYY-MM-DD');        -- PostgreSQL
SELECT DATE_FORMAT(date, '%Y-%m-%d');     -- MySQL
SELECT FORMAT(date, 'yyyy-MM-dd');         -- SQL Server

条件函数 ​

sql
-- CASE WHEN
SELECT 
    name,
    CASE 
        WHEN age < 18 THEN '未成年'
        WHEN age < 65 THEN '成年'
        ELSE '老年'
    END as age_group
FROM users;

-- COALESCE - 返回第一个非NULL值
SELECT COALESCE(phone, email, '无联系方式') FROM users;

-- NULLIF - 如果两值相等返回NULL
SELECT NULLIF(column1, 0) FROM table;

-- IF (MySQL)
SELECT IF(score > 60, '及格', '不及格') FROM students;

-- IIF (SQL Server)
SELECT IIF(score > 60, '及格', '不及格') FROM students;

三、窗口函数 ​

窗口函数使用场景：

  • 排名
  • 累计
  • 同行比较
  • 移动平均

排名函数 ​

sql
-- ROW_NUMBER - 连续排名
SELECT 
    name,
    salary,
    ROW_NUMBER() OVER (ORDER BY salary DESC) as rank
FROM employees;

-- RANK - 相同值并列，跳过序号
SELECT 
    name,
    salary,
    RANK() OVER (ORDER BY salary DESC) as rank
FROM employees;

-- DENSE_RANK - 相同值并列，不跳过序号
SELECT 
    name,
    salary,
    DENSE_RANK() OVER (ORDER BY salary DESC) as rank
FROM employees;

-- 分组排名
SELECT 
    name,
    department_id,
    salary,
    RANK() OVER (PARTITION BY department_id ORDER BY salary DESC) as dept_rank
FROM employees;

偏移函数 ​

sql
-- LAG - 获取前N行
SELECT 
    date,
    sales,
    LAG(sales, 1) OVER (ORDER BY date) as prev_sales,
    LAG(sales, 2) OVER (ORDER BY date) as prev_2_sales
FROM sales;

-- LEAD - 获取后N行
SELECT 
    date,
    sales,
    LEAD(sales, 1) OVER (ORDER BY date) as next_sales
FROM sales;

-- FIRST_VALUE - 窗口内第一个值
SELECT 
    name,
    salary,
    FIRST_VALUE(salary) OVER (PARTITION BY dept ORDER BY hire_date) as dept_first_salary
FROM employees;

-- LAST_VALUE - 窗口内最后一个值
SELECT 
    name,
    salary,
    LAST_VALUE(salary) OVER (PARTITION BY dept ORDER BY hire_date ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING)
FROM employees;

聚合窗口函数 ​

sql
-- 累计求和
SELECT 
    date,
    amount,
    SUM(amount) OVER (ORDER BY date) as cumulative_sum
FROM orders;

-- 分组累计
SELECT 
    date,
    department_id,
    amount,
    SUM(amount) OVER (PARTITION BY department_id ORDER BY date) as dept_cumulative
FROM orders;

-- 移动平均
SELECT 
    date,
    sales,
    AVG(sales) OVER (
        ORDER BY date 
        ROWS BETWEEN 2 PRECEDING AND CURRENT ROW
    ) as moving_avg_3
FROM sales;

-- 窗口函数组合使用
SELECT 
    product_id,
    date,
    sales,
    LAG(sales, 1) OVER (PARTITION BY product_id ORDER BY date) as prev_sales,
    SUM(sales) OVER (PARTITION BY product_id ORDER BY date) as cumulative_sales,
    AVG(sales) OVER (PARTITION BY product_id ORDER BY date ROWS BETWEEN 6 PRECEDING AND CURRENT ROW) as moving_avg_7
FROM sales;

函数使用对比 ​

聚合函数 vs 窗口函数 ​

sql
-- 聚合函数：返回聚合后的行
SELECT department_id, SUM(salary) 
FROM employees 
GROUP BY department_id;
-- 结果：每个部门1行

-- 窗口函数：保留所有行
SELECT name, department_id, salary, SUM(salary) OVER (PARTITION BY department_id)
FROM employees;
-- 结果：每行都有详细信息+部门总计

实际应用案例 ​

案例1：计算增长率 ​

sql
WITH sales_data AS (
    SELECT 
        date,
        revenue,
        LAG(revenue, 1) OVER (ORDER BY date) as prev_revenue
    FROM sales
)
SELECT 
    date,
    revenue,
    prev_revenue,
    (revenue - prev_revenue) * 100.0 / NULLIF(prev_revenue, 0) as growth_rate
FROM sales_data;

常见错误与陷阱 ​

sql
-- ❌ 错误1：聚合函数与普通列混用
SELECT name, SUM(salary) FROM employees;  -- 需要GROUP BY或窗口函数

-- ✅ 正确
SELECT name, SUM(salary) OVER () FROM employees;

-- ❌ 错误2：窗口函数忽略ORDER BY
SELECT LAG(salary, 1) OVER () FROM employees;  -- 需要ORDER BY

-- ✅ 正确
SELECT LAG(salary, 1) OVER (ORDER BY hire_date) FROM employees;

-- ❌ 错误3：WHERE中使用聚合函数
SELECT * FROM orders WHERE SUM(amount) > 1000;  -- 用HAVING

-- ✅ 正确
SELECT customer_id, SUM(amount) 
FROM orders 
GROUP BY customer_id 
HAVING SUM(amount) > 1000;

要点总结 ​

选择函数的原则 ​

需求选择
对一组值计算，返回单个结果聚合函数 + GROUP BY
对单个值计算，每行返回结果标量函数
需要保留所有行，同时跨行计算窗口函数（配合OVER）
需要排名ROW_NUMBER, RANK, DENSE_RANK
需要比较相邻行LAG, LEAD
需要累计计算SUM/AVG + OVER + ORDER BY

性能建议 ​

  1. 优先使用标量函数，避免在不必要场景用窗口函数
  2. 窗口函数中加上适当的 PARTITION BY
  3. 优化 ORDER BY 列，考虑索引
  4. 少用 DISTINCT（窗口内）

现在可以更准确地选择适合的函数类型。
