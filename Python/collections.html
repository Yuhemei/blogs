Python Collections 库使用指南 ​

collections 是 Python 标准库中一个非常有用的模块，提供了多种特殊的数据结构，可以替代内置的数据类型（如 dict、list、tuple），提供更强大的功能。

目录 ​

  • Counter
  • defaultdict
  • deque
  • namedtuple
  • OrderedDict
  • ChainMap

Counter ​

Counter 是一个计数器，用于统计可哈希对象的出现次数。它是 dict 的子类。

基本用法 ​

python
from collections import Counter

# 从字符串创建
counter = Counter('hello world')
print(counter)
# Counter({'l': 3, 'o': 2, 'h': 1, 'e': 1, ' ': 1, 'w': 1, 'r': 1, 'd': 1})

# 从列表创建
counter = Counter(['apple', 'banana', 'apple', 'orange', 'banana', 'apple'])
print(counter)
# Counter({'apple': 3, 'banana': 2, 'orange': 1})

# 从字典创建
counter = Counter({'red': 4, 'blue': 2})
print(counter)
# Counter({'red': 4, 'blue': 2})

常用方法 ​

python
from collections import Counter

counter = Counter(['apple', 'banana', 'apple', 'orange', 'banana', 'apple'])

# 获取元素计数
print(counter['apple'])  # 3
print(counter['grape'])  # 0 (不存在返回0，不会报错)

# 更新计数器
counter.update(['apple', 'grape'])
print(counter)
# Counter({'apple': 4, 'banana': 2, 'orange': 1, 'grape': 1})

# 获取最常见的元素
print(counter.most_common(2))
# [('apple', 4), ('banana', 2)]

# 获取所有元素
print(list(counter.elements()))
# ['apple', 'apple', 'apple', 'apple', 'banana', 'banana', 'orange', 'grape']

# 减法运算（只保留正数计数）
counter.subtract(['apple', 'banana'])
print(counter)
# Counter({'apple': 3, 'banana': 1, 'orange': 1, 'grape': 1})

# 清空
counter.clear()
print(counter)  # Counter()

defaultdict ​

defaultdict 是 dict 的一个子类，它提供了一个默认值工厂函数，当访问不存在的键时，会自动创建该键并赋予默认值。

基本用法 ​

python
from collections import defaultdict

# 默认值为列表
dd = defaultdict(list)
dd['fruits'].append('apple')
dd['fruits'].append('banana')
dd['vegetables'].append('carrot')
print(dd)
# defaultdict(<class 'list'>, {'fruits': ['apple', 'banana'], 'vegetables': ['carrot']})

# 默认值为整数
dd = defaultdict(int)
dd['count'] += 1
dd['count'] += 1
print(dd['count'])  # 2
print(dd['new_key'])  # 0 (自动创建)

# 默认值为集合
dd = defaultdict(set)
dd['tags'].add('python')
dd['tags'].add('programming')
print(dd['tags'])  # {'python', 'programming'}

python
from collections import defaultdict

# 场景1: 按类别分组
students = [
    {'name': 'Alice', 'class': 'A'},
    {'name': 'Bob', 'class': 'B'},
    {'name': 'Charlie', 'class': 'A'},
    {'name': 'David', 'class': 'B'},
]

students_by_class = defaultdict(list)
for student in students:
    students_by_class[student['class']].append(student['name'])

print(dict(students_by_class))
# {'A': ['Alice', 'Charlie'], 'B': ['Bob', 'David']}

# 场景2: 统计字符位置
text = "hello"
char_positions = defaultdict(list)
for index, char in enumerate(text):
    char_positions[char].append(index)

print(dict(char_positions))
# {'h': [0], 'e': [1], 'l': [2, 3], 'o': [4]}

# 场景3: 单词倒排索引
documents = {
    'doc1': ['python', 'programming', 'language'],
    'doc2': ['python', 'web', 'development'],
    'doc3': ['java', 'programming'],
}

inverted_index = defaultdict(set)
for doc_id, words in documents.items():
    for word in words:
        inverted_index[word].add(doc_id)

print(dict(inverted_index))
# {'python': {'doc1', 'doc2'}, 'programming': {'doc1', 'doc3'}, ...}

deque ​

deque（双端队列）是一个线程安全的、支持高效在两端添加和删除元素的数据结构。相比 list，在列表开头插入或删除元素时，deque 的性能更好（O(1) vs O(n)）。

基本用法 ​

python
from collections import deque

# 创建双端队列
dq = deque([1, 2, 3, 4])
print(dq)  # deque([1, 2, 3, 4])

# 从右侧添加元素
dq.append(5)
print(dq)  # deque([1, 2, 3, 4, 5])

# 从左侧添加元素
dq.appendleft(0)
print(dq)  # deque([0, 1, 2, 3, 4, 5])

# 从右侧删除元素
right_item = dq.pop()
print(right_item)  # 5
print(dq)  # deque([0, 1, 2, 3, 4])

# 从左侧删除元素
left_item = dq.popleft()
print(left_item)  # 0
print(dq)  # deque([1, 2, 3, 4])

高级用法 ​

python
from collections import deque

# 限制队列大小
dq = deque(maxlen=3)
dq.append(1)
dq.append(2)
dq.append(3)
print(dq)  # deque([1, 2, 3], maxlen=3)
dq.append(4)  # 自动删除最左边的元素
print(dq)  # deque([2, 3, 4], maxlen=3)

# 扩展操作
dq = deque([1, 2, 3])
dq.extend([4, 5])
print(dq)  # deque([1, 2, 3, 4, 5])

dq.extendleft([0, -1])  # 注意：会逆序添加
print(dq)  # deque([-1, 0, 1, 2, 3, 4, 5])

# 旋转
dq = deque([1, 2, 3, 4, 5])
dq.rotate(2)  # 向右旋转2位
print(dq)  # deque([4, 5, 1, 2, 3])

dq.rotate(-1)  # 向左旋转1位
print(dq)  # deque([5, 1, 2, 3, 4])

实际应用场景 ​

python
from collections import deque

# 场景1: 滑动窗口
def sliding_window_max(nums, k):
    """找出滑动窗口中的最大值"""
    dq = deque()  # 存储索引
    result = []
    
    for i, num in enumerate(nums):
        # 移除不在窗口内的元素
        while dq and dq[0] <= i - k:
            dq.popleft()
        
        # 移除所有小于当前元素的索引
        while dq and nums[dq[-1]] < num:
            dq.pop()
        
        dq.append(i)
        
        # 当窗口大小达到k时，开始记录结果
        if i >= k - 1:
            result.append(nums[dq[0]])
    
    return result

print(sliding_window_max([1, 3, -1, -3, 5, 3, 6, 7], 3))
# [3, 3, 5, 5, 6, 7]

# 场景2: 队列实现
class Queue:
    def __init__(self):
        self._queue = deque()
    
    def enqueue(self, item):
        self._queue.append(item)
    
    def dequeue(self):
        return self._queue.popleft()
    
    def is_empty(self):
        return len(self._queue) == 0

# 场景3: 最近访问记录（LRU缓存的基础）
recent_items = deque(maxlen=5)
for i in range(10):
    recent_items.append(f'item_{i}')
    print(f"添加 item_{i}: {list(recent_items)}")

namedtuple ​

namedtuple 创建一个带有命名字段的元组子类，使代码更易读和维护。

基本用法 ​

python
from collections import namedtuple

# 创建一个 Point 类
Point = namedtuple('Point', ['x', 'y'])
p = Point(1, 2)
print(p.x)  # 1
print(p.y)  # 2
print(p[0])  # 1 (仍然支持索引访问)
print(p)  # Point(x=1, y=2)

# 创建时可以使用关键字参数
p = Point(x=1, y=2)
print(p)  # Point(x=1, y=2)

# 也可以使用字符串定义字段
Point = namedtuple('Point', 'x y')
p = Point(3, 4)
print(p)  # Point(x=3, y=4)

常用方法 ​

python
from collections import namedtuple

Point = namedtuple('Point', ['x', 'y'])
p = Point(1, 2)

# _asdict(): 转换为字典
print(p._asdict())  # {'x': 1, 'y': 2}

# _replace(): 创建新的实例并替换字段
p2 = p._replace(x=10)
print(p2)  # Point(x=10, y=2)

# _make(): 从可迭代对象创建
p3 = Point._make([5, 6])
print(p3)  # Point(x=5, y=6)

# _fields: 获取所有字段名
print(Point._fields)  # ('x', 'y')

OrderedDict ​

OrderedDict 是一个有序字典，它会记住插入顺序。虽然 Python 3.7+ 的普通 dict 也保持插入顺序，但 OrderedDict 提供了额外的功能。

注意: 在 Python 3.7+ 中，普通字典已经是有序的，但 OrderedDict 仍然有其用途，特别是在需要比较顺序相等性时。

基本用法 ​

python
from collections import OrderedDict

# 创建有序字典
od = OrderedDict()
od['first'] = 1
od['second'] = 2
od['third'] = 3
print(od)  # OrderedDict([('first', 1), ('second', 2), ('third', 3)])

# 保持插入顺序
od2 = OrderedDict([('a', 1), ('b', 2), ('c', 3)])
print(od2)  # OrderedDict([('a', 1), ('b', 2), ('c', 3)])

特殊方法 ​

python
from collections import OrderedDict

od = OrderedDict([('a', 1), ('b', 2), ('c', 3)])

# move_to_end(): 移动元素到末尾
od.move_to_end('a')
print(od)  # OrderedDict([('b', 2), ('c', 3), ('a', 1)])

# move_to_end(..., last=False): 移动元素到开头
od.move_to_end('c', last=False)
print(od)  # OrderedDict([('c', 3), ('b', 2), ('a', 1)])

# popitem(last=True): 删除并返回最后一个元素
item = od.popitem()
print(item)  # ('a', 1)

# popitem(last=False): 删除并返回第一个元素
item = od.popitem(last=False)
print(item)  # ('c', 3)

实际应用场景 ​

python
from collections import OrderedDict

# 场景1: LRU Cache 实现
class LRUCache:
    def __init__(self, capacity):
        self.cache = OrderedDict()
        self.capacity = capacity
    
    def get(self, key):
        if key not in self.cache:
            return -1
        # 移动到末尾表示最近使用
        self.cache.move_to_end(key)
        return self.cache[key]
    
    def put(self, key, value):
        if key in self.cache:
            self.cache.move_to_end(key)
        self.cache[key] = value
        if len(self.cache) > self.capacity:
            # 删除最久未使用的（第一个）
            self.cache.popitem(last=False)

# 场景2: 保持配置顺序
config = OrderedDict([
    ('host', 'localhost'),
    ('port', 5432),
    ('database', 'mydb'),
    ('user', 'admin'),
])
# 按顺序处理配置项
for key, value in config.items():
    print(f"{key}: {value}")

ChainMap ​

ChainMap 将多个字典或映射链接在一起，创建一个可更新的视图。查找时按照顺序在多个映射中查找，直到找到为止。

基本用法 ​

python
from collections import ChainMap

# 创建链式映射
dict1 = {'a': 1, 'b': 2}
dict2 = {'c': 3, 'd': 4}
dict3 = {'e': 5, 'b': 10}  # 注意：'b' 会覆盖 dict1 中的 'b'

chain = ChainMap(dict1, dict2, dict3)
print(chain)  # ChainMap({'a': 1, 'b': 2}, {'c': 3, 'd': 4}, {'e': 5, 'b': 10})

# 访问元素（按顺序查找）
print(chain['a'])  # 1 (来自 dict1)
print(chain['c'])  # 3 (来自 dict2)
print(chain['b'])  # 2 (来自 dict1，因为先找到)

# 遍历所有键
print(list(chain.keys()))  # ['e', 'b', 'd', 'c', 'a']

python
from collections import ChainMap

dict1 = {'a': 1, 'b': 2}
dict2 = {'c': 3, 'd': 4}
chain = ChainMap(dict1, dict2)

# new_child(): 创建新的 ChainMap，添加新的映射
dict3 = {'e': 5}
new_chain = chain.new_child(dict3)
print(new_chain)  # ChainMap({'e': 5}, {'a': 1, 'b': 2}, {'c': 3, 'd': 4})

# parents: 获取除了第一个映射外的所有映射
print(chain.parents)  # ChainMap({'c': 3, 'd': 4})

# maps: 获取所有映射的列表
print(chain.maps)  # [{'a': 1, 'b': 2}, {'c': 3, 'd': 4}]

实际应用场景 ​

python
from collections import ChainMap

# 场景1: 配置优先级（命令行参数 > 用户配置 > 默认配置）
defaults = {'theme': 'light', 'language': 'en', 'font_size': 12}
user_config = {'theme': 'dark', 'font_size': 14}
command_line_args = {'language': 'zh'}

config = ChainMap(command_line_args, user_config, defaults)

# 高优先级配置会覆盖低优先级
print(config['theme'])    # 'dark' (来自 user_config)
print(config['language'])  # 'zh' (来自 command_line_args)
print(config['font_size']) # 14 (来自 user_config)

# 场景2: 模拟嵌套作用域
global_vars = {'x': 1, 'y': 2}
local_vars = {'x': 10, 'z': 3}

scope = ChainMap(local_vars, global_vars)
print(scope['x'])  # 10 (局部变量覆盖全局变量)
print(scope['y'])  # 2 (来自全局变量)
print(scope['z'])  # 3 (来自局部变量)

总结 ​

collections 模块提供了多种强大的数据结构：

  • Counter: 用于计数统计
  • defaultdict: 自动提供默认值的字典
  • deque: 高效的双端队列
  • namedtuple: 带命名字段的元组
  • OrderedDict: 有序字典（主要兼容性考虑）
  • ChainMap: 链式映射，用于多级配置

这些数据结构可以大大提高代码的可读性和性能，在实际开发中应该根据具体场景选择合适的工具。
